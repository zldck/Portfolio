        // Extract icon from HTML snippet
        function extractIconFromHTML() {
            const html = iconInput.value.trim();
            
            if (!html) {
                showToast('Please paste or enter an icon HTML snippet');
                return;
            }
            
            // Try to extract icon class from HTML
            const tempDiv = document.createElement('div');
            tempDiv.innerHTML = html;
            
            // Find the first <i> element
            const iconElement = tempDiv.querySelector('i');
            
            if (!iconElement) {
                showToast('No icon found in the HTML. Make sure you copied the full <i> tag.');
                return;
            }
            
            // Get the class list
            const iconClasses = iconElement.className.split(' ');
            
            // Look for FontAwesome classes
            const faClass = iconClasses.find(cls => 
                cls.startsWith('fa-') || 
                cls === 'fas' || 
                cls === 'far' || 
                cls === 'fab' || 
                cls === 'fal' || 
                cls === 'fad'
            );
            
            if (!faClass) {
                showToast('No FontAwesome icon class found. Make sure you copied from FontAwesome.');
                return;
            }
            
            // If it's a style class (fas, far, etc.), look for the actual icon class
            const iconClass = faClass.startsWith('fa-') ? faClass : 
                iconClasses.find(cls => cls.startsWith('fa-'));
            
            if (!iconClass) {
                showToast('Could not find the icon class. Make sure the HTML contains something like "fa-heart"');
                return;
            }
            
            // Select the icon
            selectIcon(iconClass);
            showToast(`Icon set to: ${iconClass}`);
        }

        // Add drag and drop functionality
        function addDragEvents(element) {
            element.addEventListener('dragstart', handleDragStart);
            element.addEventListener('dragover', handleDragOver);
            element.addEventListener('dragenter', handleDragEnter);
            element.addEventListener('dragleave', handleDragLeave);
            element.addEventListener('drop', handleDrop);
            element.addEventListener('dragend', handleDragEnd);
        }

        function handleDragStart(e) {
            draggedElement = this;
            this.classList.add('grabbing');
            e.dataTransfer.effectAllowed = 'move';
            e.dataTransfer.setData('text/html', this.innerHTML);
        }

        function handleDragOver(e) {
            e.preventDefault();
            e.dataTransfer.dropEffect = 'move';
            return false;
        }

        function handleDragEnter(e) {
            this.classList.add('drag-placeholder');
        }

        function handleDragLeave(e) {
            this.classList.remove('drag-placeholder');
        }

        function handleDrop(e) {
            e.stopPropagation();
            e.preventDefault();
            
            if (draggedElement !== this) {
                const draggedId = parseInt(draggedElement.dataset.id);
                const targetId = parseInt(this.dataset.id);
                
                // Swap projects in array
                const draggedIndex = projects.findIndex(p => p.id === draggedId);
                const targetIndex = projects.findIndex(p => p.id === targetId);
                
                [projects[draggedIndex], projects[targetIndex]] = [projects[targetIndex], projects[draggedIndex]];
                
                // Save and re-render
                saveProjectsToStorage();
                renderProjects();
                
                showToast('Projects reordered successfully!');
            }
            
            this.classList.remove('drag-placeholder');
            return false;
        }

        function handleDragEnd(e) {
            this.classList.remove('grabbing');
            document.querySelectorAll('.project-card').forEach(card => {
                card.classList.remove('drag-placeholder');
            });
        }
